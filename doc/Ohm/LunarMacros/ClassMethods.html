def lunar_fields(type, *atts)
  @lunar_fields ||= Hash.new { |h, k| h[k] = [] }

  atts.each { |att| 
    @lunar_fields[type] << att  unless @lunar_fields[type].include?(att)
  }

  @lunar_fields[type]
end
